```shell

python -m SimpleHTTPServer 9000

## 最重要的处理过程
nohup sh -c 'cat user_list.txt | while read line; do curl "127.0.0.1:8001/zeus/in/user/v1/users/$line?app_id=1000006" 2>/dev/null | jq -r ".data.user.user_id,.data.user.nick_name,.data.user.first_name,.data.user.last_name,.data.user.email,.data.user.country,.data.user.city,.data.user.interest_label" | tr "\n" ","; echo ""; done > user_info.csv' &
```
